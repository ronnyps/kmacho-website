    // Importar EffectComposer y RenderPass para el motion blur

  // Scene, Camera, Renderer setup
const scene = new THREE.Scene();
scene.fog = new THREE.Fog(0x000000, 16, 62); // Niebla negra, inicia en 50 unidades y termina en 150

const camera = new THREE.PerspectiveCamera(55, window.innerWidth / window.innerHeight, 0.1, 200); // Aumentar el rango de la cámara
camera.position.set(0, 10, 0);
camera.lookAt(0, 0, -100);

const renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });

canvasContainer.appendChild(renderer.domElement);

 

  // Add ambient light for overall brightness
  const ambientLight = new THREE.AmbientLight(0xffffff, 0.4);
  scene.add(ambientLight);

  // Add directional light to soften shadows
  const directionalLight = new THREE.DirectionalLight(0xffffff, 1.0);
  directionalLight.position.set(50, 100, 100);
  directionalLight.castShadow = false;
  
  scene.add(directionalLight);

  // Function to create a cube with very subtle rounded edges
  function createSmoothCubeGeometry(size, bevelRadius) {
    const shape = new THREE.Shape();
    const halfSize = size / 2;

    shape.moveTo(-halfSize + bevelRadius, -halfSize);
    shape.lineTo(halfSize - bevelRadius, -halfSize);
    shape.quadraticCurveTo(halfSize, -halfSize, halfSize, -halfSize + bevelRadius);
    shape.lineTo(halfSize, halfSize - bevelRadius);
    shape.quadraticCurveTo(halfSize, halfSize, halfSize - bevelRadius, halfSize);
    shape.lineTo(-halfSize + bevelRadius, halfSize);
    shape.quadraticCurveTo(-halfSize, halfSize, -halfSize, halfSize - bevelRadius);
    shape.lineTo(-halfSize, -halfSize + bevelRadius);
    shape.quadraticCurveTo(-halfSize, -halfSize, -halfSize + bevelRadius, -halfSize);

    const extrudeSettings = {
      depth: size,
      bevelEnabled: true,
      bevelSegments: 1,
      steps: 1,
      bevelSize: bevelRadius * 0.1,
      bevelThickness: bevelRadius * 0.5,
    };

    return new THREE.ExtrudeGeometry(shape, extrudeSettings);
  }

  // Grid of cubes
  const cubeGroup = new THREE.Group();
  scene.add(cubeGroup);

  const gridWidth = 35;
  const gridLength = 20;
  const cubeSize = 2.5;
  const bevelRadius = 0.1;
  const spacing = cubeSize * 1.1;

  const cubeMaterial = new THREE.MeshStandardMaterial({
  color: 0x007eff, // Tono del cristal
  metalness: 0.1, // Muy bajo o cero para cristal
  roughness: 0, // Superficie muy lisa para efecto brillante
  opacity: 0.9, // Semi-transparente
  transparent: true, // Habilitar transparencia
   envMapIntensity: 1, // Intensidad del entorno reflejado
  });

  const cubes = [];

  for (let z = 0; z < gridLength; z++) {
    for (let x = -gridWidth / 2; x <= gridWidth / 2; x++) {
      const cubeGeometry = createSmoothCubeGeometry(cubeSize, bevelRadius);
      const cube = new THREE.Mesh(cubeGeometry, cubeMaterial);
      cube.position.set(x * spacing, 0, -z * spacing);
      cube.castShadow = false;
      cube.receiveShadow = false;
      cubeGroup.add(cube);

      cubes.push({
        mesh: cube,
        initialY: cube.position.y,
        randomOffset: Math.random() * Math.PI * 2, // Random offset for wave motion
      });
    }
  }

 












  
      // Floating cards logic
      const cardGroup = new THREE.Group();
      const waveGroup = new THREE.Group(); // Grupo para las ondas
      scene.add(cardGroup);
      scene.add(waveGroup);

      // Dimensiones ajustadas de las tarjetas
      const cardWidth = 4.5; // Ancho
      const cardHeight = 4.5; // Altura
      const cardDepth = 0.2; // Profundidad para la extrusión
      const cardSpacing = 16; // Espaciado más amplio
      const cardElevation = 9; // Elevación más alta
      const cardOffsetX = gridWidth * spacing / 12; // Más cerca del centro
      const zLeftOffset = cardSpacing / 2; // Desfase adicional para las tarjetas de la izquierda

      // Cargar textura para las tarjetas
      const textureURLs = [
  'https://www.kmacho.net/wp-content/uploads/2025/01/CSS-Logo.jpg',

      // Crear material con textura
      function createCardMaterial(textureURL) {
  const textureLoader = new THREE.TextureLoader();
  const texture = textureLoader.load(
    textureURL,
    () => console.log(`Textura cargada: ${textureURL}`),
    undefined,
    (err) => console.error(`Error al cargar textura: ${err}`)
  );

  return new THREE.MeshStandardMaterial({
    map: texture,
    metalness: 0.15,
    roughness: 0.5,
  opacity: 0.9, // Semi-transparente
  transparent: true, // Habilitar transparencia
  envMapIntensity: 1, // Intensidad del entorno reflejado
  });
}


      // Función para crear la geometría extruida de las tarjetas con UV ajustados
      function createCardGeometry(width, height, depth) {
        const shape = new THREE.Shape();
        shape.moveTo(-width / 2, -height / 2); // Origen en el centro
        shape.lineTo(width / 2, -height / 2);
        shape.lineTo(width / 2, height / 2);
        shape.lineTo(-width / 2, height / 2);
        shape.lineTo(-width / 2, -height / 2);

        const extrudeSettings = {
          depth: depth,
          bevelEnabled: true,
          bevelThickness: 0.05,
          bevelSize: 0.02,
          bevelSegments: 2,
        };

        const geometry = new THREE.ExtrudeGeometry(shape, extrudeSettings);

        // Ajustar las coordenadas UV para que cubran completamente las caras superiores e inferiores
        geometry.computeBoundingBox();
        const max = geometry.boundingBox.max;
        const min = geometry.boundingBox.min;

        const uvAttribute = [];
        geometry.faceVertexUvs[0] = [];

        geometry.faces.forEach((face) => {
          const vertices = [geometry.vertices[face.a], geometry.vertices[face.b], geometry.vertices[face.c]];
          const uv = vertices.map((vertex) => {
            return new THREE.Vector2(
              (vertex.x - min.x) / (max.x - min.x),
              (vertex.y - min.y) / (max.y - min.y)
            );
          });
          geometry.faceVertexUvs[0].push(uv);
        });

        geometry.uvsNeedUpdate = true;
        return geometry;
      }

      // Crear geometría y material para la onda
      function createWave(radius) {
        const geometry = new THREE.CircleGeometry(radius, 64); // Alta segmentación para suavidad
        const material = new THREE.MeshBasicMaterial({
          color: 0x00b4ff, // Color verde
          transparent: true,
          opacity: 0.5, // Transparente al inicio
          side: THREE.DoubleSide,
          
        });

        const wave = new THREE.Mesh(geometry, material);
        wave.rotation.x = -Math.PI / 2; // Plano paralelo al suelo
        wave.scale.set(1.0, 1.0, 1.0); // Comienza pequeño
        return wave;
      }

      // Animar las ondas en cada frame
      function animateWaves() {
        waveGroup.children.forEach((wave) => {
          wave.scale.x += 0.1; // Expande en X
          wave.scale.y += 0.1; // Expande en Y
          wave.material.opacity -= 0.02; // Reduce opacidad

          if (wave.material.opacity <= 0) {
            waveGroup.remove(wave); // Eliminar ondas invisibles
          }
        });
      }

        // Crear tarjetas iniciales con textura y almacenar sus rotaciones iniciales
        const cardStates = [];
        const cardRotations = [];
        const numCards = 10; // Número de tarjetas iniciales

        for (let i = 0; i < numCards; i++) {
          // Seleccionar textura para izquierda y derecha
          const leftTextureURL = textureURLs[(i * 2) % textureURLs.length]; // Textura para la izquierda
          const rightTextureURL = textureURLs[(i * 2 + 1) % textureURLs.length]; // Textura para la derecha

          const leftCardMaterial = createCardMaterial(leftTextureURL);
          const rightCardMaterial = createCardMaterial(rightTextureURL);
          
          // Crear tarjeta izquierda
          const leftCardGeometry = createCardGeometry(cardWidth, cardHeight, cardDepth);
          const leftCard = new THREE.Mesh(leftCardGeometry, leftCardMaterial);
          leftCard.position.set(-cardOffsetX, cardElevation, -i * cardSpacing - spacing + zLeftOffset);
          const leftRotation = new THREE.Euler(Math.PI / 1.5, Math.PI / 1, -Math.PI / 1.2);
          cardRotations.push(leftRotation);
          leftCard.rotation.copy(leftRotation);
          cardGroup.add(leftCard);

          // Crear tarjeta derecha
          const rightCardGeometry = createCardGeometry(cardWidth, cardHeight, cardDepth);
          const rightCard = new THREE.Mesh(rightCardGeometry, rightCardMaterial);
          rightCard.position.set(cardOffsetX, cardElevation, -i * cardSpacing - spacing);
          const rightRotation = new THREE.Euler(Math.PI / 1.5, Math.PI / 1, -Math.PI / 1.2);
          cardRotations.push(rightRotation);
          rightCard.rotation.copy(rightRotation);
          cardGroup.add(rightCard);

          // Inicializar estados de las tarjetas
          cardStates.push(
            {
              floatingState: { position: { ...leftCard.position }, rotation: { ...leftRotation } },
              onFloorState: null,
            },
            {
              floatingState: { position: { ...rightCard.position }, rotation: { ...rightRotation } },
              onFloorState: null,
            }
          );
        }


      // Animación de las tarjetas (incluye regeneración infinita y efecto de onda)
      function animateCards() {
        cardGroup.children.forEach((card, index) => {
          if (card.position.z < camera.position.z + 25 && card.position.z > camera.position.z - 25) {
            card.position.x += (0 - card.position.x) * 0.1;
            card.position.y += (2 - card.position.y) * 0.1;
            card.rotation.x += (Math.PI / 2 - card.rotation.x) * 0.1;
            card.rotation.y += (Math.PI / 1 - card.rotation.y) * 0.1;
            card.rotation.z += (Math.PI / -1 - card.rotation.z) * 0.1;


            
      // Generar una onda al finalizar la animación de la tarjeta
      if (!card.userData.waveGenerated) {
        const positionThreshold = 0.2; // Tolerancia para la posición
        const rotationThreshold = 0.2; // Tolerancia para la rotación

        const isPositionReached =
          Math.abs(card.position.x - 0) < positionThreshold &&
          Math.abs(card.position.y - 2) < positionThreshold &&
          Math.abs(card.position.z - card.position.z) < positionThreshold;

        const isRotationReached =
          Math.abs(card.rotation.x - Math.PI / 2) < rotationThreshold &&
          Math.abs(card.rotation.y - Math.PI / 1) < rotationThreshold &&
          Math.abs(card.rotation.z - Math.PI / -1) < rotationThreshold;

        if (isPositionReached && isRotationReached) {
          const wave = createWave(1.5); // Radio inicial de la onda
          wave.position.set(0, 2.00, card.position.z); // Bajo la tarjeta
          waveGroup.add(wave);

          card.userData.waveGenerated = true; // Marcar como onda generada
        }
      }

          } else {
            card.position.y =
              cardElevation + Math.sin(Date.now() * 0.001 + index) * 0.5;
            card.userData.waveGenerated = false; // Reiniciar la onda si se mueve
          }

          if (card.position.z > camera.position.z + 10) {
            card.position.z -= numCards * cardSpacing;
            card.position.x = card.position.x < 0 ? -cardOffsetX : cardOffsetX;
            card.position.y = cardElevation;
            const originalRotation = cardRotations[index];
            card.rotation.copy(originalRotation);
            card.userData.waveGenerated = false; // Reiniciar onda
          }
        });
        animateWaves(); // Animar las ondas
      }
























gsap.registerPlugin(ScrollTrigger);

// Parámetros generales
let baseCameraSpeed = 0.05;
let cameraSpeed = baseCameraSpeed;
const maxElevation = 55;
const lateralThreshold = spacing * 2;
const zRange = 40;
const elevationSpeed = 0.01;
const returnSpeed = 0.1;
let animationPaused = true; // Iniciar pausado

// Variables para seguimiento del mouse
let mouseX = 0, mouseY = 0;
const cameraRotationIntensity = 0.03;

// Actualizar coordenadas del mouse
window.addEventListener('mousemove', (event) => {
  const { innerWidth, innerHeight } = window;
  mouseX = (event.clientX / innerWidth) * 2 - 1;
  mouseY = -(event.clientY / innerHeight) * 2 + 1;
});

// Configuración de ScrollTrigger
ScrollTrigger.create({
  trigger: "#background-canvas-asd", // El canvas
  start: "top-=200 top", // Cuando el canvas entra al viewport
  end: "bottom+=100 bottom", // Cuando el canvas sale completamente

  onEnter: () => {
    animationPaused = false; // Reanudar animación
    cameraSpeed = baseCameraSpeed; // Velocidad normal
    console.log("Canvas visible: animaciones reanudadas");

   

    // Detener cualquier efecto de wiggle al entrar
    gsap.killTweensOf(camera.rotation);
    gsap.to(camera.rotation, {
      x: 0,
      y: 0,
      z: 0,
      duration: 0.5,
      ease: "power2.out",
    });
  },

  onLeave: () => {
    animationPaused = true; // Pausar animación
    cameraSpeed = baseCameraSpeed * 4; // Aumentar más la velocidad de la cámara al salir
    console.log("Canvas fuera del viewport: cámara acelerada con efecto wiggle activado");

    // Aplicar efecto wiggle a la cámara
    const wiggleIntensity = 0.02; // Intensidad del efecto wiggle
    gsap.to(camera.rotation, {
      x: `+=${wiggleIntensity}`, // Oscilación en X
      y: `+=${wiggleIntensity}`, // Oscilación en Y
      z: `+=${wiggleIntensity / 2}`, // Oscilación en Z
      repeat: -1, // Repetición infinita
      yoyo: true, // Efecto de ida y vuelta
      duration: 0.5,
      ease: "sine.inOut",
    });

    // Las tarjetas caen al nivel del suelo en su posición y rotación específicas
    cardGroup.children.forEach((card, index) => {
    cardStates[index].onFloorState = {
      position: { x: 0, y: 2, z: card.position.z },
      rotation: { x: Math.PI / 2, y: 0, z: 0 },
    };

    gsap.to(card.position, {
      x: cardStates[index].onFloorState.position.x,
      y: cardStates[index].onFloorState.position.y,
      z: cardStates[index].onFloorState.position.z,
      duration: 0.5,
      ease: "power2.out",
    });

    gsap.to(card.rotation, {
      x: cardStates[index].onFloorState.rotation.x,
      y: cardStates[index].onFloorState.rotation.y,
      z: cardStates[index].onFloorState.rotation.z,
      duration: 0.5,
      ease: "power2.out",
    });
  });

    // Cubos vuelven a su posición inicial o flotan aleatoriamente
    const floatingCubes = [];
    while (floatingCubes.length < 5) {
      const randomIndex = Math.floor(Math.random() * cubes.length);
      if (!floatingCubes.includes(randomIndex)) {
        floatingCubes.push(randomIndex);
      }
    }

    cubes.forEach((cubeData, index) => {
      const { mesh } = cubeData;

      if (floatingCubes.includes(index)) {
        const randomHeight = 5 + Math.random() * 5; // Altura aleatoria entre 5 y 10
        cubeData.floatingHeight = randomHeight; // Guardar la altura flotante para regeneración
        gsap.to(mesh.position, {
          y: randomHeight,
          duration: 1,
          ease: "power2.out",
        });
        gsap.to(mesh.rotation, {
          x: "+=180", // Rotar continuamente en X
          y: "+=180", // Rotar continuamente en Y
          repeat: -1,
          duration: 150,
          ease: "linear",
        });
        mesh.userData.floating = true;
      } else {
        cubeData.floatingHeight = 0;
        gsap.to(mesh.position, {
          y: 0,
          duration: 0.5,
          ease: "power2.out",
        });
        mesh.userData.floating = false;
      }
    });
  },

  onEnterBack: () => {
    console.log("Canvas reingresó al viewport");

    cardGroup.children.forEach((card, index) => {
      // Asegurarse de restaurar el estado flotante
      const floatingState = cardStates[index].floatingState;

      gsap.to(card.position, {
        x: floatingState.position.x,
        y: floatingState.position.y,
  
        duration: 0.5,
        ease: "power2.out",
      });

      const rotation = cardRotations[index];
    gsap.to(card.rotation, {
      x: rotation.x,
      y: rotation.y,
      z: rotation.z,
      duration: 0.5,
      ease: "power2.out",
    });
    });


    cubes.forEach((cubeData) => {
      const { mesh, initialY } = cubeData;
      if (mesh.userData.floating) {
        gsap.killTweensOf(mesh.rotation);
        gsap.to(mesh.rotation, {
          x: 0,
          y: 0,
          duration: 0.5,
          ease: "power2.out",
        });
        gsap.to(mesh.position, {
          y: initialY,
          duration: 0.5,
          ease: "power2.out",
        });
        mesh.userData.floating = false;
      }
    });

    // Detener efecto de wiggle al regresar
    gsap.killTweensOf(camera.rotation);
    gsap.to(camera.rotation, {
      x: 0,
      y: 0,
      z: 0,
      duration: 0.5,
      ease: "power2.out",
    });

    animationPaused = false;
    cameraSpeed = baseCameraSpeed;
    console.log("Canvas reingresó al viewport");
  },

  onLeaveBack: () => {
    animationPaused = true;
    cameraSpeed = baseCameraSpeed * 3; // Acelerar cámara al salir hacia arriba
    console.log("Canvas salió del viewport por la parte superior");
  },
});

// Función de animación
function animate() {
  requestAnimationFrame(animate);

  // Rotación de la cámara en función del mouse
  const targetRotationX = mouseY * cameraRotationIntensity;
  const targetRotationY = mouseX * cameraRotationIntensity;

  camera.rotation.x += (targetRotationX - camera.rotation.x) * 0.1;
  camera.rotation.y += (targetRotationY - camera.rotation.y) * 0.1;

  // Movimiento de la cámara
  camera.position.z -= cameraSpeed;

  // Reubicar cubos que han pasado la cámara
  cubes.forEach((cubeData) => {
    const { mesh } = cubeData;

    if (mesh.position.z > camera.position.z + 1) {
      mesh.position.z -= gridLength * spacing;
      mesh.position.y = cubeData.floatingHeight || cubeData.initialY;
    }
  });

  // Reubicar tarjetas que pasan la cámara para crear un loop infinito
  cardGroup.children.forEach((card, index) => {
  if (card.position.z > camera.position.z + 10) {
    // Obtener el estado flotante almacenado
    const floatingState = cardStates[index].floatingState;

    // Reubicar en el bucle infinito
    card.position.x = floatingState.position.x; // Usar posición flotante almacenada
    card.position.y = floatingState.position.y;

    // Restaurar rotación inicial
    card.rotation.set(
      floatingState.rotation.x,
      floatingState.rotation.y,
      floatingState.rotation.z
    );
  }
});


  // Animación de los cubos
  if (!animationPaused) {
    cubes.forEach((cubeData) => {
      const { mesh, initialY, randomOffset } = cubeData;

      const distanceToCameraZ = Math.abs(mesh.position.z - camera.position.z);
      if (distanceToCameraZ < zRange) {
        if (Math.abs(mesh.position.x) > lateralThreshold) {
          const lateralFactor = Math.abs(mesh.position.x) / (gridWidth * spacing);
          const elevationFactor = 1 - distanceToCameraZ / zRange;
          const wave = Math.sin(Date.now() * 0.001 + randomOffset) * 2;
          const targetY =
            initialY + lateralFactor * elevationFactor * maxElevation + wave;

          mesh.position.y += (targetY - mesh.position.y) * elevationSpeed;
        } else {
          mesh.position.y += (initialY - mesh.position.y) * elevationSpeed;
        }
      } else {
        mesh.position.y += (initialY - mesh.position.y) * elevationSpeed;
      }
    });

    animateCards();
  }

  // Renderizar la escena
  renderer.render(scene, camera);
}

// Manejar redimensionamiento de la ventana
window.addEventListener("resize", () => {
  camera.aspect = window.innerWidth / window.innerHeight;
  camera.updateProjectionMatrix();
  renderer.setSize(window.innerWidth, window.innerHeight);
});

// Iniciar la animación
animate();



</script>
